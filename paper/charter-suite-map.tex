\textbf{Files in this release:}
\begin{itemize}[itemsep=0.2em, topsep=0.2em, leftmargin=2em]
  \item \textbf{Core Charter (Constitutional Core)} — \textit{normative, canonical}. Source of truth for rights claims and binding obligations.
  \item \textbf{Technical Standard} — \textit{normative, subordinate}. Operational requirements, procedures, and audit artifacts implementing the Core.
  \item \textbf{Commentary \& Rationale} — \textit{non-normative}. Interludes, design rationale, historical parallels, and scenario walkthroughs.
\end{itemize}

\textbf{Precedence and legal force:}
\begin{itemize}[itemsep=0.2em, topsep=0.2em, leftmargin=2em]
  \item \textbf{Precedence:} Core > Technical Standard > Commentary.
  \item \textbf{Non-derogation:} The Technical Standard may not reduce protections guaranteed by the Core.
  \item \textbf{Version lock:} The Technical Standard is expected to evolve faster than the Core; each release should declare the Core version range it is compatible with.
\end{itemize}

\textbf{How to cite:}
\begin{itemize}[itemsep=0.2em, topsep=0.2em, leftmargin=2em]
  \item Cite \textbf{Core} for rights claims and binding obligations.
  \item Cite \textbf{Technical Standard} for operational/compliance claims and auditable procedures.
  \item Cite \textbf{Commentary} for rationale, interpretation, and scenario analysis.
\end{itemize}

}
